import random           #  
import matplotlib       
import matplotlib.pyplot as plt     #  

# Binomial Random Variable B~(10,0.5)
exp3 = []
for i in range(1024):         #  1024 
    count = 0                   # HEAD    
    for j in range(10):         # n = 10
        if random.randrange(0,2)==1:       # 50%
            count += 1
    exp3.append(count)          # exp3 1024     . ( 0~10)

plt.hist(exp3)
plt.show()



comb=[]     # calculation combination by dynamic programming
for i in range(11):
    comb.append([])
    for j in range(11):
        comb[i].append(0)

for i in range(11):
    comb[i][0] = 1

for i in range(1,11):
    for j in range(1,11):
        comb[i][j] = comb[i-1][j-1]+comb[i-1][j]
#DP end


def calc(ph,n,k):
    return comb[n][k]*ph**k*(1-ph)**(n-k) # B~(n,k)  

poslist = []
for j in range(10001): # p=0~1  0.0001  
    phsum = 0
    for i in range(1024):
        phsum += calc(j/10000,10,exp3[i])*(1/1024) # 1024 sample  
    poslist.append(phsum)

maxpos = 0          # ML  *10000
maxcalc = 0         #   Likelihood(P)
for i in range(10001):  # find max
    if maxcalc < poslist[i]:
        maxpos = i
        maxcalc = poslist[i]

print(' p hat by ML Estimation: ',maxpos/10000)
print('  likelihood: ',maxcalc)

import random   # 
import matplotlib
import matplotlib.pyplot as plt # 
from numpy import exp #  

#   
samples = 3000

exp4 = []       #    
exp42 = []      # PMF   
for i in range(1001):       # n=1000 PMF x  0~1000.
    exp42.append(0)
for i in range(0,samples):  # samples     
    count = 0               # HEAD 
    for j in range(1000):   # n=1000
        if random.randrange(0,100)==1:  # 0~99 1  1/100 p=0.01
            count += 1
    exp4.append(count)      #  
    exp42[count] += 1       # PMF( x count (y++)) 

for i in range(1001):   # x=0~1000  (count/totalcount)
    exp42[i] /= samples

#histogram
plt.hist(exp4)
plt.show()

#PMF Plot
plt.plot(exp42)
plt.xlim((0,30))    #  x  0~30  (     0)
plt.show()

def poissonPos(ld,k):   # lambda k 
    fact = 1
    for i in range(2,k+1): #  
        fact *= i
    return ((ld**k)*exp(-ld)/fact)

pp = [] # poisson possibilities, k=0~30

for i in range(31):
    pp.append(poissonPos(10,i))

plt.plot(pp)
