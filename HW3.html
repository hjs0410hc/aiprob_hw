import random       # 난수 생성
import matplotlib  
import matplotlib.pyplot as plt     # 그래프 plot


# P(HEAD) = 0.5인 경우
exp1=[]             #idx가 x축(동전 던진 횟수), value가 y축(head가 나온 횟수)
for i in range(1,101):
    count = 0
    for j in range(i):                      # i번 던진다 (i는 1~100)
        if random.randrange(0,2) == 1:      # 0,1 둘 중 하나만 나오는 랜덤에서 1을 고르면 50%
            count+=1
    exp1.append(count/i*100)                # 총 던진 횟수에서 head가 나온 횟수의 percentage

plt.plot(range(1,101),exp1)                 # 그래프 작성



# P(HEAD) = 0.6 인 경우
exp2=[]            #exp1과 동일
for i in range(1,101):
    count = 0
    for j in range(i):
        if random.randrange(0,5) < 3:       # 0,1,2,3,4 총 5개의 난수에서 0,1,2를 고르면 60%.
            count+=1
    exp2.append(count/i*100)

plt.plot(range(1,101),exp2,'r')             # 빨간색으로 그래프 작성
plt.ylim(0,100)                             # y축 범위 (0,100)으로 한정
plt.show()

Discussion (problem 1-c)¶

  • 파란색이 P(HEAD)=0.5, 빨간색이 P(HEAD)=0.6
  • 동전을 던진 횟수가 늘어날 수록 head의 비율이 P(HEAD)로 수렴하는 것을 그래프로 확인할 수 있다. Law of large numbers가 성립하는 것을 실험적으로 확인했다.
  • Law of Large Numbers : 표본의 수가 늘어날 수록 표본의 평균(eg. HEAD/COUNT)이 모집단의 평균(P(HEAD))과 가까울 가능성이 높음

import random           # 난수 생성
import matplotlib       
import matplotlib.pyplot as plt     #그래프 작성에 사용

# Binomial Random Variable B~(10,0.5)
exp3 = []
for i in range(1024):         # 이항분포를 1024번 표본
    count = 0                   # HEAD가 나온 횟수 저장 변수
    for j in range(10):         # n = 10
        if random.randrange(0,2)==1:       # 50%
            count += 1
    exp3.append(count)          # exp3에 1024개의 이항분포 표본 각각의 결과를 저장. (결과값은 0~10임)

plt.hist(exp3)
plt.show()



comb=[]     # calculation combination by dynamic programming
for i in range(11):
    comb.append([])
    for j in range(11):
        comb[i].append(0)

for i in range(11):
    comb[i][0] = 1

for i in range(1,11):
    for j in range(1,11):
        comb[i][j] = comb[i-1][j-1]+comb[i-1][j]
#DP end


def calc(ph,n,k):
    return comb[n][k]*ph**k*(1-ph)**(n-k) # B~(n,k) 확률 계산

poslist = []
for j in range(10001): # p=0~1 까지 0.0001 단위로 계산
    phsum = 0
    for i in range(1024):
        phsum += calc(j/10000,10,exp3[i])*(1/1024) # 1024개의 sample이므로 각각을 나눠줌
    poslist.append(phsum)

maxpos = 0          # ML로 계산된 확률*10000
maxcalc = 0         # 해당 확률의 Likelihood(P)
for i in range(10001):  # find max
    if maxcalc < poslist[i]:
        maxpos = i
        maxcalc = poslist[i]

print('계산된 p hat by ML Estimation: ',maxpos/10000)
print('이 때 likelihood: ',maxcalc)

계산된 p hat by ML Estimation:  0.4982
이 때 likelihood:  0.18045061487140981

import random   #난수 생성
import matplotlib
import matplotlib.pyplot as plt #그래프 작성
from numpy import exp # 자연지수함수 사용

# 이항분포 표본 수
samples = 3000

exp4 = []       # 표본 저장하기 위한 배열
exp42 = []      # PMF 저장하기 위한 배열
for i in range(1001):       # n=1000이므로 PMF x축의 값은 0~1000여야한다.
    exp42.append(0)
for i in range(0,samples):  # samples 번 만큼 이항분포의 표본 추출
    count = 0               # HEAD의 횟수
    for j in range(1000):   # n=1000
        if random.randrange(0,100)==1:  # 0~99에서 1이 나오면 1/100이므로 p=0.01
            count += 1
    exp4.append(count)      # 표본 저장
    exp42[count] += 1       # PMF(그래프 x값에 count 추가(y++)) 저장

for i in range(1001):   # x=0~1000를 확률화 (count/totalcount)
    exp42[i] /= samples

#histogram
plt.hist(exp4)
plt.show()

#PMF Plot
plt.plot(exp42)
plt.xlim((0,30))    # 임의로 x축 값을 0~30으로 제한 (이 후로는 값이 대체로 전부 0임)
plt.show()

# 포아송 분포 계산 함수
def poissonPos(ld,k):   # lambda와 k 입력
    fact = 1
    for i in range(2,k+1): # 팩토리얼 계산
        fact *= i
    return ((ld**k)*exp(-ld)/fact)

pp = [] # poisson possibilities, k=0~30

for i in range(31):
    pp.append(poissonPos(10,i))

plt.plot(pp)
plt.show()

문제 2-c & 2-d¶

  • d. c의 pmf를 그렸다. (위 3개 그래프 중 2번째) 포아송 분포의 그래프도 그렸다. (마지막 그래프) 이항분포에서 p가 작고 n이 큰 경우 lambda=np인 포아송 분포와 매우 비슷해짐을 확인했다.
